Operand       | Value
--------------+------
%rax          | 0x100
0x104         | 0xAB
$0x108        | 0x108
(%rax)        | 0xFF
4(%rax)       | 0xAB
9(%rax,%rdx)  | 0x11
260(%rcx,%rdx | 0x13
0xFC(,%rcx,4) | 0xFF
(%rax,%rdx,4) | 0x11

4(%rax): 4 + 0x100 = 0x104 -> 0xAB
9(%rax,%rdx): 9 + 0x100 + 0x3 = 0x10C -> 0x11
260(%rcx,%rdx: 260 + 0x1 + 0x3 = 264 = 0x108 -> 0x13
0xFC(,%rcx,4): 0xFC + 0x1*4 = 0xFC + 0x4 = 0x100 -> 0xFF
(%rax,%rdx,4): 0x100 + 0x3*4 = 0x100 + 0xC = 0x10C -> 0x11
